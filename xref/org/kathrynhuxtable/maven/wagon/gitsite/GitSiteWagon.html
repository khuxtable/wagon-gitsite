67  /**
68   * Wagon provider to deploy site documentation to GitHub's pages system.
69   *
70   * <p>This should do more or less the following, but doesn't because it doesn't
71   * actually delete old files.</p>
72   *
73   * <pre>
74   * mkdir ${checkoutDirectory}
75   * cd ${checkoutDirectory}
76   * git init
77   * git remote add origin ${gitRepoUrl}
78   * git pull origin refs/heads/${siteBranch}
79   * <replace the contents of the checkout directory, except for the .git subdirectory, with the site docs>
80   * git add .
81   * git commit -a -m "Wagon: Deploying site to repository"
82   * git push origin master:${siteBranch}
83   * rm -Rf ${checkoutDirectory}
84   * </pre>
85   *
86   * @plexus.component role="org.apache.maven.wagon.Wagon" role-hint="gitsite"
87   *                   instantiation-strategy="per-lookup"
88   * @author           <a href="kathryn@kathrynhuxtable.org">Kathryn Huxtable</a>
89   * @author           <a href="brett@apache.org">Brett Porter</a>
90   * @author           <a href="evenisse@apache.org">Emmanuel Venisse</a>
91   * @author           <a href="carlos@apache.org">Carlos Sanchez</a>
92   * @author           Jason van Zyl

180     /**

182      *
183      * @throws ConnectionException
184      */
185     public void openConnectionInternal() throws ConnectionException {
186         if (checkoutDirectory == null) {
187             checkoutDirectory = createCheckoutDirectory();
188         }
189 
190         if (checkoutDirectory.exists()) {
191             removeCheckoutDirectory();
192         }
193 
194         checkoutDirectory.mkdirs();
195     }

197     /**
198      * Create the checkout directory.
199      *
200      * @return the File representing the checkout directory.
201      */
202     private File createCheckoutDirectory() {
203         File checkoutDirectory;
204 
205         DecimalFormat fmt = new DecimalFormat("#####");
206 
207         Random rand = new Random(System.currentTimeMillis() + Runtime.getRuntime().freeMemory());
208 
209         synchronized (rand) {
210             do {
211                 checkoutDirectory = new File(System.getProperty("java.io.tmpdir"),
212                                              "wagon-scm" + fmt.format(Math.abs(rand.nextInt())) + ".checkout");
213             } while (checkoutDirectory.exists());
214         }
215 
216         return checkoutDirectory;
217     }

219     /**
220      * Remove (delete) the checkout directory.
221      *
222      * @throws ConnectionException if unable to clean up the checkout directory.
223      */
224     private void removeCheckoutDirectory() throws ConnectionException {
225         if (checkoutDirectory == null) {
226             return; // Silently return.
227         }
228 
229         try {
230             FileUtils.deleteDirectory(checkoutDirectory);
231         } catch (IOException e) {
232             throw new ConnectionException("Unable to cleanup checkout directory", e);
233         }
234     }

236     /**
237      * Get the SCM repository from the URL.
238      *
239      * @param  url the URL.
240      *
241      * @return the SCM repository.
242      *
243      * @throws ScmRepositoryException     if an SCM error occurs.
244      * @throws NoSuchScmProviderException if there is no matching provider for
245      *                                    the URL.
246      */
247     private ScmRepository getScmRepository(String url) throws ScmRepositoryException, NoSuchScmProviderException {
248         String username = null;
249 
250         String password = null;
251 
252         String privateKey = null;
253 
254         String passphrase = null;
255 
256         if (authenticationInfo != null) {
257             username = authenticationInfo.getUserName();
258 
259             password = authenticationInfo.getPassword();
260 
261             privateKey = authenticationInfo.getPrivateKey();
262 
263             passphrase = authenticationInfo.getPassphrase();
264         }
265 
266         ScmRepository scmRepository = getScmManager().makeScmRepository(url);
267 
268         ScmProviderRepository providerRepository = scmRepository.getProviderRepository();
269 
270         if (StringUtils.isNotEmpty(username)) {
271             providerRepository.setUser(username);
272         }
273 
274         if (StringUtils.isNotEmpty(password)) {
275             providerRepository.setPassword(password);
276         }
277 
278         if (providerRepository instanceof ScmProviderRepositoryWithHost) {
279             ScmProviderRepositoryWithHost providerRepo = (ScmProviderRepositoryWithHost) providerRepository;
280 
281             if (StringUtils.isNotEmpty(privateKey)) {
282                 providerRepo.setPrivateKey(privateKey);
283             }
284 
285             if (StringUtils.isNotEmpty(passphrase)) {
286                 providerRepo.setPassphrase(passphrase);
287             }
288         }
289 
290         return scmRepository;
291     }

293     /**
294      * Configure and perform the check-in process.
295      *
296      * @param  scmProvider   the SCM provider.
297      * @param  scmRepository the SCM repository.
298      * @param  msg           the commit message.
299      *
300      * @throws ScmException
301      */
302     private void checkIn(ScmProvider scmProvider, ScmRepository scmRepository, String msg) throws ScmException {
303         CommandParameters parameters = new CommandParameters();
304 
305         parameters.setScmVersion(CommandParameter.SCM_VERSION, new ScmBranch(siteBranch));
306 
307         parameters.setString(CommandParameter.MESSAGE, msg);
308 
309         ScmResult result = (CheckInScmResult) executeCommand((GitExeScmProvider) scmProvider, new GitSiteCheckInCommand(),
310                                                              scmRepository.getProviderRepository(),
311                                                              new ScmFileSet(checkoutDirectory), parameters);
312 
313         checkScmResult(result);
314     }

316     /**
317      * Configure and perform the check-out process.
318      *
319      * <p>Returns the relative path to targetName in the checkout dir. If the
320      * targetName already exists in the scm, this will be the empty string.</p>
321      *
322      * @param  scmProvider   the SCM provider.
323      * @param  scmRepository the SCM repository.
324      * @param  targetName    the check-out directory.
325      * @param  resource      the resource.
326      *
327      * @return the relative path to targetName in the check-out directory.
328      *
329      * @throws TransferFailedException
330      */
331     private String checkOut(ScmProvider scmProvider, ScmRepository scmRepository, String targetName, Resource resource)
332         throws TransferFailedException {
333         checkoutDirectory = createCheckoutDirectory();
334 
335         Stack<String> stack = new Stack<String>();
336 
337         String target = targetName;
338 
339         // Totally ignore scmRepository parent stuff since that is not supported by all SCMs.
340         // Instead, assume that that url exists. If not, then that's an error.
341         // Check whether targetName, which is a relative path into the scm, exists.
342         // If it doesn't, check the parent, etc.
343 
344         try {
345             while (target.length() > 0
346                     && !scmProvider.list(scmRepository, new ScmFileSet(new File("."), new File(target)), false, (ScmVersion) null)
347                     .isSuccess()) {
348                 stack.push(getFilename(target));
349                 target = getDirname(target);
350             }
351         } catch (ScmException e) {
352             fireTransferError(resource, e, TransferEvent.REQUEST_PUT);
353 
354             throw new TransferFailedException("Error listing repository: " + e.getMessage(), e);
355         }
356 
357         // ok, we've established that target exists, or is empty.
358         // Check the resource out; if it doesn't exist, that means we're in the svn repo url root,
359         // and the configuration is incorrect. We will not try repo.getParent since most scm's don't
360         // implement that.
361 
362         try {
363             scmRepository = getScmRepository(getRepository().getUrl());
364 
365             CommandParameters parameters = new CommandParameters();
366 
367             parameters.setScmVersion(CommandParameter.SCM_VERSION, new ScmBranch(siteBranch));
368 
369             parameters.setString(CommandParameter.RECURSIVE, "false");
370 
371             CheckOutScmResult ret = (CheckOutScmResult) executeCommand((GitExeScmProvider) scmProvider, new GitSiteCheckOutCommand(),
372                                                                        scmRepository.getProviderRepository(),
373                                                                        new ScmFileSet(new File(checkoutDirectory, "")), parameters);
374 
375             checkScmResult(ret);
376         } catch (ScmException e) {
377             fireTransferError(resource, e, TransferEvent.REQUEST_PUT);
378 
379             throw new TransferFailedException("Error checking out: " + e.getMessage(), e);
380         }
381 
382         // now create the subdirs in target, if it's a parent of targetName
383 
384         String relPath = "";
385 
386         while (!stack.isEmpty()) {
387             String p = (String) stack.pop();
388 
389             relPath += p + "/";
390 
391             File newDir = new File(checkoutDirectory, relPath);
392 
393             if (!newDir.mkdirs()) {
394                 throw new TransferFailedException("Failed to create directory " + newDir.getAbsolutePath() + "; parent should exist: "
395                                                   + checkoutDirectory);
396             }
397 
398             try {
399                 addFiles(scmProvider, scmRepository, checkoutDirectory, relPath);
400             } catch (ScmException e) {
401                 fireTransferError(resource, e, TransferEvent.REQUEST_PUT);
402 
403                 throw new TransferFailedException("Failed to add directory " + newDir + " to working copy", e);
404             }
405         }
406 
407         return relPath;
408     }

410     /**
411      * Add a file or directory to a SCM repository. If it's a directory all its
412      * contents are added recursively.
413      *
414      * <p>TODO this is less than optimal, SCM API should provide a way to add a
415      * directory recursively</p>
416      *
417      * @param  scmProvider   SCM provider
418      * @param  scmRepository SCM repository
419      * @param  basedir       local directory corresponding to scmRepository
420      * @param  scmFilePath   path of the file or directory to add, relative to
421      *                       basedir
422      *
423      * @return the number of files added.
424      *
425      * @throws ScmException
426      */
427     private int addFiles(ScmProvider scmProvider, ScmRepository scmRepository, File basedir, String scmFilePath) throws ScmException {
428         int addedFiles = 0;
429 
430         File scmFile = new File(basedir, scmFilePath);
431 
432         if (scmFilePath.length() != 0) {
433             AddScmResult result = scmProvider.add(scmRepository, new ScmFileSet(basedir, new File(scmFilePath)));
434 
435             /*
436              * TODO dirty fix to work around files with property svn:eol-style=native if a file has that property, first
437              * time file is added it fails, second time it succeeds the solution is check if the scm provider is svn and
438              * unset that property when the SCM API allows it
439              */
440             if (!result.isSuccess()) {
441                 result = scmProvider.add(scmRepository, new ScmFileSet(basedir, new File(scmFilePath)));
442             }
443 
444             addedFiles = result.getAddedFiles().size();
445         }
446 
447         String reservedScmFile = scmProvider.getScmSpecificFilename();
448 
449         if (scmFile.isDirectory()) {
450             File[] files = scmFile.listFiles();
451 
452             for (int i = 0; i < files.length; i++) {
453                 if (reservedScmFile != null && !reservedScmFile.equals(files[i].getName())) {
454                     addedFiles += addFiles(scmProvider, scmRepository, basedir,
455                                            (scmFilePath.length() == 0 ? "" : scmFilePath + "/") + files[i].getName());
456                 }
457             }
458         }
459 
460         return addedFiles;
461     }

474     /**
475      * @see org.apache.maven.wagon.AbstractWagon#connect(org.apache.maven.wagon.repository.Repository,
476      *      org.apache.maven.wagon.authentication.AuthenticationInfo,
477      *      org.apache.maven.wagon.proxy.ProxyInfoProvider)
478      */
479     public void connect(Repository repository, AuthenticationInfo authenticationInfo, ProxyInfoProvider proxyInfoProvider)
480         throws ConnectionException, AuthenticationException {
481         String url = repository.getUrl();
482 
483         if (url.startsWith("gitsite:")) {
484             url = url.substring(8);
485             int index = url.indexOf(':');
486 
487             if (index > -1) {
488                 siteBranch = url.substring(index + 1);
489                 url        = url.substring(0, index);
490             } else {
491                 siteBranch = "gh-pages";
492             }
493 
494             url = "scm:git:ssh://" + url;
495             repository.setUrl(url);
496         }
497 
498         super.connect(repository, authenticationInfo, proxyInfoProvider);
499     }

508     /**
509      * @see org.apache.maven.wagon.AbstractWagon#putDirectory(java.io.File, java.lang.String)
510      */
511     public void putDirectory(File sourceDirectory, String destinationDirectory) throws TransferFailedException,
512         ResourceDoesNotExistException, AuthorizationException {
513         if (!sourceDirectory.isDirectory()) {
514             throw new IllegalArgumentException("Source is not a directory: " + sourceDirectory);
515         }
516 
517         Resource target = new Resource(destinationDirectory);
518 
519         firePutInitiated(target, sourceDirectory);
520 
521         try {
522             ScmRepository scmRepository = getScmRepository(getRepository().getUrl());
523 
524             target.setContentLength(sourceDirectory.length());
525             target.setLastModified(sourceDirectory.lastModified());
526 
527             firePutStarted(target, sourceDirectory);
528 
529             String msg = "Wagon: Deploying " + sourceDirectory.getName() + " to repository";
530 
531             ScmProvider scmProvider = getScmProvider(scmRepository.getProvider());
532 
533             String checkoutTargetName = sourceDirectory.isDirectory() ? destinationDirectory : getDirname(destinationDirectory);
534             String relPath            = checkOut(scmProvider, scmRepository, checkoutTargetName, target);
535 
536             File newCheckoutDirectory = new File(checkoutDirectory, relPath);
537 
538             File scmFile = new File(newCheckoutDirectory, sourceDirectory.isDirectory() ? "" : getFilename(destinationDirectory));
539 
540             boolean fileAlreadyInScm = scmFile.exists();
541 
542             if (!scmFile.equals(sourceDirectory)) {
543                 if (sourceDirectory.isDirectory()) {
544                     FileUtils.copyDirectoryStructure(sourceDirectory, scmFile);
545                 } else {
546                     FileUtils.copyFile(sourceDirectory, scmFile);
547                 }
548             }
549 
550             if (!fileAlreadyInScm || scmFile.isDirectory()) {
551                 int addedFiles = addFiles(scmProvider, scmRepository, newCheckoutDirectory,
552                                           sourceDirectory.isDirectory() ? "" : scmFile.getName());
553 
554                 if (!fileAlreadyInScm && addedFiles == 0) {
555                     throw new ScmException("Unable to add file to SCM: " + scmFile + "; see error messages above for more information");
556                 }
557             }
558 
559             checkIn(scmProvider, scmRepository, msg);
560         } catch (ScmException e) {
561             e.printStackTrace();
562             fireTransferError(target, e, TransferEvent.REQUEST_GET);
563 
564             System.exit(1);
565             throw new TransferFailedException("Error interacting with SCM: " + e.getMessage(), e);
566         } catch (IOException e) {
567             fireTransferError(target, e, TransferEvent.REQUEST_GET);
568 
569             throw new TransferFailedException("Error interacting with SCM: " + e.getMessage(), e);
570         }
571 
572         if (sourceDirectory.isFile()) {
573             postProcessListeners(target, sourceDirectory, TransferEvent.REQUEST_PUT);
574         }
575 
576         firePutCompleted(target, sourceDirectory);
577     }

617     /**
618      * Get the file list for the resource.
619      *
620      * @param  resourcePath the resource path.
621      *
622      * @return a List&lt;String&gt; with filenames/directories at the
623      *         resourcepath.
624      *
625      * @throws TransferFailedException
626      * @throws ResourceDoesNotExistException
627      * @throws AuthorizationException
628      *
629      * @see    org.apache.maven.wagon.AbstractWagon#getFileList(java.lang.String)
630      */
631     public List<String> getFileList(String resourcePath) throws TransferFailedException, ResourceDoesNotExistException,
632         AuthorizationException {
633         try {
634             ScmRepository repository = getScmRepository(getRepository().getUrl());
635             ScmProvider   provider   = getScmProvider(repository.getProvider());
636             ListScmResult result     = provider.list(repository,
637                                                      new ScmFileSet(new File("."), new File(resourcePath)), false, (ScmVersion) null);
638 
639             if (!result.isSuccess()) {
640                 throw new ResourceDoesNotExistException(result.getProviderMessage());
641             }
642 
643             List<String> files = new ArrayList<String>();
644 
645             for (ScmFile f : getListScmResultFiles(result)) {
646                 files.add(f.getPath());
647                 System.out.println("LIST FILE: " + f + " (path=" + f.getPath() + ")");
648             }
649 
650             return files;
651         } catch (ScmException e) {
652             throw new TransferFailedException("Error getting filelist from SCM", e);
653         }
654     }
